%--- Multiple choice (4) -------------------------------------
\begin{tcolorbox}[ugconceptbox, title={Part I --- Multiple Choice \hfill\normalfont 4 questions $\times$ 2 pts = 8 pts}]

\begin{enumerate}[leftmargin=1.8em, label=\textbf{Q\arabic*.}, itemsep=10pt]

\item \tiertag{tier1col}{T1}
  A quantum source prepares spin-up with probability $\tfrac{1}{2}$ and spin-down
  with probability $\tfrac{1}{2}$. Which statement best describes the resulting state?

  \begin{enumerate}[label=(\Alph*), itemsep=2pt]
    \item It is the superposition $|\psi\rangle=\tfrac{1}{\sqrt{2}}(|\uparrow\rangle+|\downarrow\rangle)$.
    \item It is a mixed state described by $\hat{\rho}=\tfrac{1}{2}|\uparrow\rangle\langle\uparrow|+\tfrac{1}{2}|\downarrow\rangle\langle\downarrow|$.
    \item It is a pure state with coefficients $c_\uparrow=c_\downarrow=\tfrac{1}{2}$.
    \item It cannot be described by quantum mechanics because no single ket exists.
  \end{enumerate}

\item \tiertag{tier2col}{T2}
  For a pure state $|\psi\rangle$, which of the following equals the density operator $\hat{\rho}$?

  \begin{enumerate}[label=(\Alph*), itemsep=2pt]
    \item $|\psi\rangle/\langle\psi|\psi\rangle$
    \item $|\psi\rangle\langle\psi|$
    \item $\langle\psi|\psi\rangle\hat{\mathbb{1}}$
    \item $\hat{\mathbb{1}}-|\psi\rangle\langle\psi|$
  \end{enumerate}

\item \tiertag{tier2col}{T2}
  The purity $\gamma=\operatorname{Tr}(\hat{\rho}^2)$ of a pure state equals:

  \begin{enumerate}[label=(\Alph*), itemsep=2pt]
    \item $0$
    \item $1/d$ where $d$ is the Hilbert-space dimension
    \item $1$
    \item $\operatorname{Tr}(\hat{\rho})$
  \end{enumerate}

\item \tiertag{tier3col}{T3}
  Which of the following matrices is a \emph{valid} density operator for a qubit?

  \begin{enumerate}[label=(\Alph*), itemsep=2pt]
    \item $\begin{pmatrix}1/2 & 1 \\ 1 & 1/2\end{pmatrix}$
    \item $\begin{pmatrix}3/4 & 1/4 \\ 1/4 & 1/4\end{pmatrix}$
    \item $\begin{pmatrix}1/2 & 0 \\ 0 & -1/2\end{pmatrix}$
    \item $\begin{pmatrix}3/4 & i/8 \\ -i/8 & 1/4\end{pmatrix}$
  \end{enumerate}

\end{enumerate}
\end{tcolorbox}

%--- True/False + justify (4) ---------------------------------
\begin{tcolorbox}[ugconceptbox, title={Part II --- True / False + one-sentence justification \hfill\normalfont 4 $\times$ 2 pts = 8 pts}]

\begin{enumerate}[leftmargin=1.8em, label=\textbf{Q\arabic*.}, itemsep=10pt, start=5]

\item \tiertag{tier1col}{T1}
  \textbf{True or False:} Multiplying a ket by a global phase $e^{i\theta}$
  changes the density operator $\hat{\rho}=|\psi\rangle\langle\psi|$.\\[3pt]
  \textit{Justify in one sentence.}

  \vspace{2cm}

\item \tiertag{tier2col}{T2}
  \textbf{True or False:} For any density operator $\hat{\rho}$,
  $\operatorname{Tr}(\hat{\rho})=1$ and $\operatorname{Tr}(\hat{\rho}^2)\le 1$.\\[3pt]
  \textit{Justify in one sentence.}

  \vspace{2cm}

\item \tiertag{tier2col}{T2}
  \textbf{True or False:} The expectation value formula
  $\langle\hat{A}\rangle=\operatorname{Tr}(\hat{\rho}\hat{A})$ reduces to
  $\langle\psi|\hat{A}|\psi\rangle$ when $\hat{\rho}=|\psi\rangle\langle\psi|$.\\[3pt]
  \textit{Justify in one sentence.}

  \vspace{2cm}

\item \tiertag{tier3col}{T3}
  \textbf{True or False:} A qubit state with Bloch vector satisfying
  $|\mathbf{r}|=1/2$ is a pure state.\\[3pt]
  \textit{Justify in one sentence, referencing purity.}

  \vspace{2cm}

\end{enumerate}
\end{tcolorbox}

%--- Short answer (2) ----------------------------------------
\begin{tcolorbox}[ugconceptbox, title={Part III --- Short Answer \hfill\normalfont 2 $\times$ 2 pts = 4 pts}]

\begin{enumerate}[leftmargin=1.8em, label=\textbf{Q\arabic*.}, itemsep=10pt, start=9]

\item \tiertag{tier3col}{T3}
  Two distinct options for a qubit mixed state are:
  (A)~$\hat{\rho}_A=\tfrac{3}{4}|\uparrow\rangle\langle\uparrow|+\tfrac{1}{4}|\downarrow\rangle\langle\downarrow|$
  and
  (B)~$\hat{\rho}_B=\tfrac{1}{2}|{+}\rangle\langle{+}|+\tfrac{1}{2}|{-}\rangle\langle{-}|$
  where $|\pm\rangle=\tfrac{1}{\sqrt{2}}(|\uparrow\rangle\pm|\downarrow\rangle)$.
  Which has higher purity? Compute both purities.

  \vspace{2.5cm}

\item \tiertag{tier3col}{T3}
  Explain in 2--3 sentences why a subsystem of a bipartite pure entangled
  state can itself be mixed. Your answer should reference the partial trace.

  \vspace{2.5cm}

\end{enumerate}
\end{tcolorbox}

\medskip\noindent
\textbf{Answers:} Q1~(B); Q2~(B); Q3~(C); Q4~(D);
Q5~False ($e^{i\theta}$ cancels in $|\psi\rangle\langle\psi|$);
Q6~True; Q7~True; Q8~False ($|\mathbf{r}|=1/2$ gives $\gamma=5/8<1$, mixed);
Q9~$\gamma_A=10/16=0.625$, $\gamma_B=1/2=0.5$, so A has higher purity;
Q10~Partial trace over B sums out B's degrees of freedom, yielding a mixed reduced state if the global state is entangled.
\end{tcolorbox}

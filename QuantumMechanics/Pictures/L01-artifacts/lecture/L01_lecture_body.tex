\begin{tcolorbox}[notebox]
\textbf{Lecture summary.}
This lecture introduces the density operator $\hat{\rho}$ as the fundamental
descriptor of quantum states --- pure, mixed, and reduced --- replacing the
single-ket picture that fails for open systems and statistical ensembles.
Students learn to construct $\hat{\rho}$, verify its defining properties
(Hermiticity, unit trace, positive semi-definiteness), compute expectation
values via the trace, quantify mixedness via purity, and extract subsystem
states via the partial trace.
The density matrix unifies the entire Module~I.3 treatment of dynamics and
perturbation theory, and is the gateway to the open-system content of L10.
\end{tcolorbox}

  \item \textbf{Compute} (simple): given $p_1,p_2$ with $p_1+p_2=1$, write down

\subsection{L01.3 --- Motivation: When Is a Ket Not Enough?}
\label{sec:L01.3.1}
%-------------------------------------------------------------

\begin{tcolorbox}[goldbox]
\textbf{Main result.}
A single ket $|\psi\rangle\in\mathcal{H}$ describes only \emph{pure} quantum
states. Three physically distinct situations require a more general description:
(i)~statistical ensembles of preparation procedures, (ii)~quantum subsystems of
entangled composite systems, and (iii)~open systems coupled to unobserved environments.
The density operator $\hat{\rho}$ handles all three cases uniformly.
\end{tcolorbox}

\subsubsection*{Three Motivating Scenarios}

\textbf{Scenario 1: Statistical ensemble.}
A preparation machine selects $|\uparrow\rangle$ with probability $p$ and
$|\downarrow\rangle$ with probability $1-p$.
No single ket encodes this classical uncertainty, yet all predictions follow
from $\hat{\rho}=p|\uparrow\rangle\langle\uparrow|+(1-p)|\downarrow\rangle\langle\downarrow|$.

\textbf{Scenario 2: Subsystem of entangled pair.}
Consider two qubits in the Bell state
$|\Phi^+\rangle=\tfrac{1}{\sqrt{2}}(|00\rangle+|11\rangle)$.
Qubit $A$ alone has no valid ket description: its predictions are captured only
by $\hat{\rho}_A=\operatorname{Tr}_B(|\Phi^+\rangle\langle\Phi^+|)=\tfrac{1}{2}\hat{\mathbb{1}}$.

\textbf{Scenario 3: Open system.}
An atom interacts with a photon field. After tracing over field degrees of
freedom, the atom's state is mixed, and its evolution is generated not by the
Schr\"odinger equation but by a master equation for $\hat{\rho}$ (see L10).

\instructornote{Draw the Bloch sphere on the board: T1 students focus on the
sphere picture; T2--T3 on computing $\hat{\rho}$; T4--T5 on the convex geometry
of the state space.}

\subsection{L01.4 --- Constructing the Density Operator}
\label{sec:L01.3.2}
%-------------------------------------------------------------

\begin{tcolorbox}[goldbox]
\textbf{Main result.}
Any valid quantum state is described by a \emph{density operator} $\hat{\rho}$
satisfying three conditions: (i)~$\hat{\rho}^\dagger=\hat{\rho}$ (Hermitian),
(ii)~$\operatorname{Tr}(\hat{\rho})=1$ (unit trace), (iii)~$\hat{\rho}\ge 0$
(positive semi-definite). Conversely, any operator satisfying these three conditions
is a valid density operator.
\end{tcolorbox}

\subsubsection*{Definitions}

\begin{description}[leftmargin=1.5em]
  \item[\textbf{Pure state:}]
    $\hat{\rho}=|\psi\rangle\langle\psi|$\quad (rank-1 projector; $\hat{\rho}^2=\hat{\rho}$)

  \item[\textbf{Statistical mixture:}]
    $\hat{\rho}=\displaystyle\sum_i p_i\,|\psi_i\rangle\langle\psi_i|$,\quad
    $p_i\ge 0$,\quad $\displaystyle\sum_i p_i=1$

  \item[\textbf{Reduced density operator:}]
    $\hat{\rho}_A=\operatorname{Tr}_B(\hat{\rho}_{AB})
     =\displaystyle\sum_j\bigl(\hat{\mathbb{1}}_A\otimes\langle j|_B\bigr)
     \hat{\rho}_{AB}
     \bigl(\hat{\mathbb{1}}_A\otimes|j\rangle_B\bigr)$

  \item[\textbf{Expectation value:}]
    $\langle\hat{A}\rangle=\operatorname{Tr}(\hat{\rho}\hat{A})$
    \quad (basis-independent by cyclicity of trace)
\end{description}

\subsubsection*{Core Derivation: Proof of the Three Properties}
\label{sec:L01.3.2.proof}

\tiertag{tier3col}{Tier 3} \tiertag{tier4col}{Tier 4} \tiertag{tier5col}{Tier 5}

\medskip\noindent
For the mixture $\hat{\rho}=\sum_i p_i|\psi_i\rangle\langle\psi_i|$
with $p_i\ge 0$, $\sum_i p_i=1$:

\begin{enumerate}[leftmargin=2em]
  \item \textbf{Hermitian:}
    $\hat{\rho}^\dagger=\sum_i p_i(|\psi_i\rangle\langle\psi_i|)^\dagger
     =\sum_i p_i|\psi_i\rangle\langle\psi_i|=\hat{\rho}$.
    \hfill$\checkmark$

  \item \textbf{Unit trace:}
    $\operatorname{Tr}(\hat{\rho})=\sum_i p_i\,\operatorname{Tr}(|\psi_i\rangle\langle\psi_i|)
     =\sum_i p_i\langle\psi_i|\psi_i\rangle=\sum_i p_i=1$.
    \hfill$\checkmark$

  \item \textbf{Positive semi-definite:}
    For any $|\phi\rangle$,
    $\langle\phi|\hat{\rho}|\phi\rangle
     =\sum_i p_i|\langle\phi|\psi_i\rangle|^2\ge 0$,
    since each $p_i\ge 0$ and $|\langle\phi|\psi_i\rangle|^2\ge 0$.
    \hfill$\checkmark$
\end{enumerate}

\begin{tcolorbox}[goldbox]
\textbf{Worked Example 1: Spin-$\tfrac{1}{2}$ mixture.}\\
A source produces $|\uparrow\rangle$ with probability $\tfrac{3}{4}$ and
$|\downarrow\rangle$ with probability $\tfrac{1}{4}$.

\medskip
In the $\{|\uparrow\rangle,|\downarrow\rangle\}$ basis:
\begin{equation}
\hat{\rho}=\frac{3}{4}\begin{pmatrix}1&0\\0&0\end{pmatrix}
+\frac{1}{4}\begin{pmatrix}0&0\\0&1\end{pmatrix}
=\begin{pmatrix}3/4&0\\0&1/4\end{pmatrix}.
\label{eq:L01.1}
\end{equation}

\textit{Check:} $\operatorname{Tr}(\hat{\rho})=1$\checkmark;\quad
$\hat{\rho}\ge 0$ (both eigenvalues positive)\checkmark;\quad
$\hat{\rho}^\dagger=\hat{\rho}$\checkmark.
\end{tcolorbox}

%-------------------------------------------------------------
\subsection{L01.5 --- Purity and the Bloch Sphere}
\label{sec:L01.3.3}
%-------------------------------------------------------------

\begin{tcolorbox}[goldbox]
\textbf{Main result.}
The \emph{purity} $\gamma=\operatorname{Tr}(\hat{\rho}^2)$ lies in $[1/d,1]$
for a $d$-dimensional Hilbert space.
$\gamma=1$ if and only if $\hat{\rho}$ is a pure state;
$\gamma=1/d$ for the maximally mixed state $\hat{\rho}=\hat{\mathbb{1}}/d$.
\end{tcolorbox}

\subsubsection*{Proof: $\operatorname{Tr}(\hat{\rho}^2)\le 1$}

\tiertag{tier3col}{Tier 3}

\medskip\noindent
By spectral decomposition, $\hat{\rho}=\sum_k \lambda_k|k\rangle\langle k|$
with $\lambda_k\ge 0$, $\sum_k\lambda_k=1$.
Then $\operatorname{Tr}(\hat{\rho}^2)=\sum_k\lambda_k^2$.
By the Cauchy--Schwarz inequality (or by convexity of $t\mapsto t^2$):
\begin{equation}
\sum_k\lambda_k^2\le\Bigl(\sum_k\lambda_k\Bigr)^2=1.
\label{eq:L01.2}
\end{equation}
Equality holds iff all probability is concentrated on one eigenvalue,
i.e.\ $\hat{\rho}=|k_0\rangle\langle k_0|$ is rank-1 (pure state).

\subsubsection*{Bloch Sphere (qubit, $d=2$)}

\tiertag{tier2col}{Tier 2} \tiertag{tier3col}{Tier 3}

\medskip\noindent
Any qubit density operator can be written
\begin{equation}
\hat{\rho}=\tfrac{1}{2}(\hat{\mathbb{1}}+\mathbf{r}\cdot\hat{\boldsymbol{\sigma}}),
\quad \mathbf{r}=(r_x,r_y,r_z)\in\mathbb{R}^3,
\label{eq:L01.3}
\end{equation}
where $\hat{\boldsymbol{\sigma}}=(\hat{\sigma}_x,\hat{\sigma}_y,\hat{\sigma}_z)$
are the Pauli matrices.
The conditions $\hat{\rho}\ge 0$ and $\operatorname{Tr}(\hat{\rho})=1$ are
equivalent to $|\mathbf{r}|\le 1$.
Purity: $\gamma=\tfrac{1}{2}(1+|\mathbf{r}|^2)$.
Pure states: $|\mathbf{r}|=1$ (Bloch sphere surface).
Maximally mixed state: $\mathbf{r}=0$, $\gamma=\tfrac{1}{2}$.

\begin{tcolorbox}[goldbox]
\textbf{Worked Example 2: Expectation value via trace.}\\
For the state in Example~1 and $\hat{A}=\hat{\sigma}_z=\text{diag}(1,-1)$:
\begin{align}
\langle\hat{\sigma}_z\rangle
&=\operatorname{Tr}(\hat{\rho}\hat{\sigma}_z)
 =\operatorname{Tr}\!\begin{pmatrix}3/4&0\\0&1/4\end{pmatrix}\begin{pmatrix}1&0\\0&-1\end{pmatrix}
 =\operatorname{Tr}\!\begin{pmatrix}3/4&0\\0&-1/4\end{pmatrix}\notag\\
&=\tfrac{3}{4}-\tfrac{1}{4}=\tfrac{1}{2}.
\label{eq:L01.4}
\end{align}
Purity: $\gamma=\operatorname{Tr}(\hat{\rho}^2)=(3/4)^2+(1/4)^2=9/16+1/16=10/16=5/8<1$\quad (mixed).
\end{tcolorbox}

%-------------------------------------------------------------
\subsection{L01.6 --- Partial Trace and Reduced Density Operators}
\label{sec:L01.3.4}
%-------------------------------------------------------------

\begin{tcolorbox}[goldbox]
\textbf{Main result.}
For a composite system $AB$ with state $\hat{\rho}_{AB}\in\mathcal{B}(\mathcal{H}_A\otimes\mathcal{H}_B)$,
the \emph{reduced density operator} of subsystem $A$ is
$\hat{\rho}_A=\operatorname{Tr}_B(\hat{\rho}_{AB})$.
This is the unique operator such that
$\operatorname{Tr}(\hat{\rho}_A\hat{A})=\operatorname{Tr}(\hat{\rho}_{AB}(\hat{A}\otimes\hat{\mathbb{1}}_B))$
for all $\hat{A}$ on $\mathcal{H}_A$.
\end{tcolorbox}

\subsubsection*{Computation}

Choose an orthonormal basis $\{|j\rangle_B\}$ for $\mathcal{H}_B$.
Then:
\begin{equation}
\hat{\rho}_A=\operatorname{Tr}_B(\hat{\rho}_{AB})
=\sum_j\bigl(\hat{\mathbb{1}}_A\otimes{}_B\langle j|\bigr)\,
\hat{\rho}_{AB}\,
\bigl(\hat{\mathbb{1}}_A\otimes|j\rangle_B\bigr).
\label{eq:L01.5}
\end{equation}

\begin{tcolorbox}[goldbox]
\textbf{Worked Example 3: Bell state partial trace.}\\
$|\Phi^+\rangle=\tfrac{1}{\sqrt{2}}(|00\rangle+|11\rangle)$.
Full state: $\hat{\rho}_{AB}=|\Phi^+\rangle\langle\Phi^+|$.

Using basis $\{|0\rangle_B,|1\rangle_B\}$:
\begin{align}
\hat{\rho}_A
&={}_B\langle 0|\hat{\rho}_{AB}|0\rangle_B+{}_B\langle 1|\hat{\rho}_{AB}|1\rangle_B\notag\\
&=\tfrac{1}{2}|0\rangle_A\langle 0|+\tfrac{1}{2}|1\rangle_A\langle 1|
 =\tfrac{1}{2}\hat{\mathbb{1}}_A.
\label{eq:L01.6}
\end{align}
\textit{Interpretation:} Qubit $A$ is maximally mixed ($\gamma_A=1/2$) even though
the global two-qubit state is pure --- a signature of entanglement.
\end{tcolorbox}

\subsubsection*{von Neumann Entropy and Entanglement}

\tiertag{tier4col}{Tier 4} \tiertag{tier5col}{Tier 5}

\begin{equation}
S(\hat{\rho}_A)=-\operatorname{Tr}(\hat{\rho}_A\ln\hat{\rho}_A)
=-\sum_k\lambda_k\ln\lambda_k\ge 0.
\label{eq:L01.7}
\end{equation}
For the Bell state: $S(\hat{\rho}_A)=\ln 2$ (maximum for a qubit).
For a product state $\hat{\rho}_{AB}=\hat{\rho}_A\otimes\hat{\rho}_B$:
$S(\hat{\rho}_A)$ is the entropy of $\hat{\rho}_A$ alone (no entanglement contribution).

\subsubsection*{Purification}
\tiertag{tier4col}{Tier 4}

\medskip\noindent
Given any $\hat{\rho}_A$ on $\mathcal{H}_A$ with spectral decomposition
$\hat{\rho}_A=\sum_k\lambda_k|k\rangle_A\langle k|$, define the purification
\begin{equation}
|\Psi\rangle_{AB}=\sum_k\sqrt{\lambda_k}|k\rangle_A|k\rangle_B\in\mathcal{H}_A\otimes\mathcal{H}_B.
\label{eq:L01.8}
\end{equation}
Then $\operatorname{Tr}_B(|\Psi\rangle\langle\Psi|)=\hat{\rho}_A$ by construction.
This is the Schmidt decomposition of $|\Psi\rangle$.

    Section~3.5 introduces the density matrix in the wave-mechanics language;
    Section~3.6 covers measurement and mixed states. Directly supports

  \item \tiertag{tier2col}{T2}\
    \textbf{Townsend, J.~S.}

  \item \tiertag{tier3col}{T3}\

    The primary T3 reference. Section~3.4 (density operator), 3.5

  \item \tiertag{tier3col}{T3}\
    \textbf{von Neumann, J.}
    \textit{Mathematical Foundations of Quantum Mechanics}, Ch.~IV.

  \item \tiertag{tier3col}{T3}\

    Description of states in quantum mechanics by density matrix and operator techniques.
    \textit{Rev.~Mod.~Phys.}~\textbf{29}, 74--93 (1957). \recommended\ [\textbf{P}]\\

    \textit{States, Effects, and Operations}, Springer Lecture Notes~190, 1983.

    \textit{Commun.\ Math.\ Phys.}~\textbf{48}, 119--130 (1976). \essential\ [\textbf{P}]\\

    The T5 anchor for ORQ~4--5. Preview for L10.

  \item \tiertag{tier5col}{T5}\

    \textit{J.\ Math.\ Phys.}~\textbf{17}, 821--825 (1976). \essential\ [\textbf{P}]\\

    Complements Lindblad (1976); both papers together are the definitive
    reference for CPJG~Project~5 and ORQ~4.

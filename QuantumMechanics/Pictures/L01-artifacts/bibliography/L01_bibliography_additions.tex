  \textit{This file documents all references used in the L01 artifact suite,

%=============================================================
\subsection*{Tier 1--3 (Undergraduate) --- Core References}
%=============================================================

\begin{enumerate}[leftmargin=2em, label={\textbf{[\arabic*]}}, itemsep=10pt]

\item $\bigstar$
  \textbf{Griffiths, D.~J.\ \& Schroeter, D.~F.}
  \textit{Introduction to Quantum Mechanics}, 3rd edition.
  Cambridge University Press, 2018.
  \textit{Sections}: \S\,3.5 (density matrix), \S\,3.6 (mixed states and measurement).\\
  \textit{Used in}: SCQU Q4--Q6, SPSU Part~A, CCQU Q2; lecture body Section~3.2.\\
  \textit{Master bib key}: \texttt{Griffiths2018}.\\
  \textit{Notes}: Primary T2 reference. The $\sigma$-algebra treatment of probability
  is absent but the physical intuition is excellent for beginning undergraduates.

\item
  \textbf{Townsend, J.~S.}
  \textit{A Modern Approach to Quantum Mechanics}, 2nd edition.
  University Science Books, 2012.
  \textit{Chapter}: 7 (density matrices for spin systems).\\
  \textit{Used in}: SPSU Part~B (interpretation problems), SCQU Q1--Q3.\\
  \textit{Master bib key}: \texttt{Townsend2012}.\\
  \textit{Notes}: Complements Griffiths for T2 students who prefer a spin-first approach.

\item $\bigstar$
  \textbf{Sakurai, J.~J.\ \& Napolitano, J.}
  \textit{Modern Quantum Mechanics}, 3rd edition.
  Cambridge University Press, 2020.
  \textit{Sections}: \S\,3.4 (density operator), \S\,3.5 (ensembles), \S\,3.6 (time evolution of $\hat{\rho}$).\\
  \textit{Used in}: CPSU Problems 1--5, CCQU Q1--Q4, lecture body Sections 3.2--3.4; SPSU Part~C.\\
  \textit{Master bib key}: \texttt{Sakurai2020}.\\
  \textit{Notes}: The primary T3 reference. The Bloch-sphere derivation in \S\,3.4 is
  the cleanest available at this level. Problem 3.9--3.12 are excellent CPSU warm-ups.

\item
  \textbf{Penrose, R.}
  \textit{The Emperor's New Mind}.
  Oxford University Press, 1989. Chapters 5--6.\\
  \textit{Used in}: SPJU Project~J3 (conceptual background for T1).\\
  \textit{Master bib key}: \texttt{Penrose1989}.\\
  \textit{Notes}: Non-technical; appropriate as pre-lecture reading for T1 students.
  Chapter~6 contains an unusually clear discussion of the distinction between
  superposition and mixture for a lay audience.

\item $\bigstar$
  \textbf{Fano, U.}
  Description of states in quantum mechanics by density matrix and operator techniques.
  \textit{Reviews of Modern Physics}, \textbf{29}(1), 74--93, 1957.\\
  \textit{DOI}: \href{https://doi.org/10.1103/RevModPhys.29.74}{10.1103/RevModPhys.29.74}\\
  \textit{Used in}: SPJU Project~J3, CPJU Project~CJ1, WRQ~1; lecture body Section~3.1.\\
  \textit{Master bib key}: \texttt{Fano1957} \textbf{[P]}.\\
  \textit{Notes}: Classic pedagogical review written for atomic physicists.
  Section~IV (spin density matrix and polarisation) is the key reading for J3.
  The ``polarisation vector'' is the Bloch vector in modern language.

\item
  \textbf{von Neumann, J.}
  \textit{Mathematical Foundations of Quantum Mechanics}.
  Princeton University Press, 1955 (English translation of 1932 German edition).
  Chapter~IV (statistical operator).\\
  \textit{Used in}: WRQ~1, CCQG Q4; lecture body Section~3.1.\\
  \textit{Master bib key}: \texttt{vonNeumann1955} \textbf{[M]}.\\
  \textit{Notes}: The original introduction of the density operator (``statistische Operator'').
  Chapter~IV is dense but accessible at T3 with guidance. Historical context for WRQ~1.

\end{enumerate}

\subsection*{Tier 4--5 (Graduate) --- Core References}
%=============================================================

\begin{enumerate}[leftmargin=2em, label={\textbf{[\arabic*]}}, itemsep=10pt, resume]

\item $\bigstar$
  \textbf{Weinberg, S.}
  \textit{Lectures on Quantum Mechanics}, 2nd edition.
  Cambridge University Press, 2015. Chapter~3.\\
  \textit{Used in}: SCQG Q1--Q5, CCQG Q5--Q6, CPJG Project~CG2; lecture body Section~5.\\
  \textit{Master bib key}: \texttt{Weinberg2015}.\\
  \textit{Notes}: Cleanest graduate treatment of the density operator.
  Chapter~3.3--3.4 (mixed states and decoherence) directly support CPSU~5 and CPJG~CG2.
  Weinberg's notation ($\rho$ for the density matrix element) differs from the
  hat notation used in this programme; students should be warned.

\item $\bigstar$
  \textbf{Ballentine, L.~E.}
  \textit{Quantum Mechanics: A Modern Development}, 2nd edition.
  World Scientific, 2014. Chapter~2 (state preparation and density matrix).\\
  \textit{Used in}: SCQG Q2--Q3, CCQG Q1--Q2, CPSG Problems 1--2, WRQ~2.\\
  \textit{Master bib key}: \texttt{Ballentine2014}.\\
  \textit{Notes}: The statistical-ensemble interpretation is developed rigorously.
  Chapter~2.3--2.4 are the key sections. Ballentine distinguishes proper and improper
  mixtures more carefully than most texts (relevant to WRQ~2).

\item $\bigstar$
  \textbf{Nielsen, M.~A.\ \& Chuang, I.~L.}
  \textit{Quantum Computation and Quantum Information}.
  Cambridge University Press, 2010.
  \textit{Sections}: \S\,2.4 (density operators), \S\,2.5 (Schmidt decomposition),
  \S\,8.2--8.3 (quantum operations and Kraus).\\
  \textit{Used in}: SCQG Q6--Q7, CCQG Q7, CPSG Problems 1--3, SPJG Projects~SG1,
  SG2; WRQ~3--4; SPJU Project~CJ1.\\
  \textit{Master bib key}: \texttt{NielsenChuang2010}.\\
  \textit{Notes}: The single most comprehensive reference at T4 level.
  Section~2.4 is essential; \S\,8.2 (operator-sum representation) and \S\,8.3
  (Kraus rank) support CPSG~Problems~3--5 and SPJG~SG1.

\item $\bigstar$
  \textbf{Reed, M.\ \& Simon, B.}
  \textit{Methods of Modern Mathematical Physics}, Vol.~I: Functional Analysis.
  Academic Press, 1972. Chapter~VI (compact and trace-class operators).\\
  \textit{Used in}: SCQG Q8--Q10, CPSG Problem~5, CPJG Project~CG1.\\
  \textit{Master bib key}: \texttt{ReedSimon1972} \textbf{[M]}.\\
  \textit{Notes}: The mathematical reference for T5 students.
  Theorem~VI.24 (trace-class operators form a Banach space) is used directly in
  CPSG~5(b). Proposition~VI.26 (density operators) and the discussion of
  Schatten $p$-norms are key.

\item
  \textbf{Kraus, K.}
  \textit{States, Effects, and Operations: Fundamental Notions of Quantum Theory}.
  Springer Lecture Notes in Physics, Vol.~190, 1983.\\
  \textit{Used in}: CPSG Problem~5, CPJG Projects~CG2, CG3; WRQ~3; ORQ~3.\\
  \textit{Master bib key}: \texttt{Kraus1983} \textbf{[M]}.\\
  \textit{Notes}: Introduces the Kraus representation theorem and CPTP maps.
  Chapter~II (operations) and III (effects) are essential for CG2.

\item $\bigstar$
  \textbf{Lindblad, G.}
  On the generators of quantum dynamical semigroups.
  \textit{Communications in Mathematical Physics}, \textbf{48}(2), 119--130, 1976.\\
  \textit{DOI}: \href{https://doi.org/10.1007/BF01608499}{10.1007/BF01608499}\\
  \textit{Used in}: CPSG Problem~4, CPJG Project~CG2; WRQ~4; ORQ~4; lecture body Section~4.\\
  \textit{Master bib key}: \texttt{Lindblad1976} \textbf{[P]}.\\
  \textit{Notes}: The GKSL paper. The proof of the generator form (their Theorem~1)
  is rigorous but uses GNS construction; T5 students should work through it with
  Reed \& Simon Vol.~II as background.

\item $\bigstar$
  \textbf{Gorini, V., Kossakowski, A.\ \& Sudarshan, E.~C.~G.}
  Completely positive dynamical semigroups of $N$-level systems.
  \textit{Journal of Mathematical Physics}, \textbf{17}(5), 821--825, 1976.\\
  \textit{DOI}: \href{https://doi.org/10.1063/1.522979}{10.1063/1.522979}\\
  \textit{Used in}: CPSG Problem~4, CPJG Project~CG2; WRQ~4; ORQ~4.\\
  \textit{Master bib key}: \texttt{GKS1976} \textbf{[P]}.\\
  \textit{Notes}: Independent derivation of the Lindblad form for finite-dimensional
  systems. Together with Lindblad (1976), these are the definitive references for
  CPJG~CG2 and ORQ~4. The finite-dimensional proof in GKS is more accessible than
  Lindblad's infinite-dimensional version.

\end{enumerate}

\subsection*{Research Track --- Additional References}
%=============================================================

\begin{enumerate}[leftmargin=2em, label={\textbf{[\arabic*]}}, itemsep=10pt, resume]

\item
  \textbf{Holevo, A.~S.}
  \textit{Quantum Systems, Channels, Information: A Mathematical Introduction}.
  De~Gruyter, 2012.\\
  \textit{Used in}: WRQ~3, CPJG Projects~CG2, CG3.\\
  \textit{Master bib key}: \texttt{Holevo2012} \textbf{[M]}.\\
  \textit{Notes}: Information-theoretic treatment of quantum channels from a
  mathematical physics perspective. Chapters~6--7 (capacity and entropy) support WRQ~4.

\item
  \textbf{Wilde, M.~M.}
  \textit{Quantum Information Theory}, 2nd edition.
  Cambridge University Press, 2017. Chapters~3--5 (entropy, channels).\\
  \textit{Available free}: \href{https://arxiv.org/abs/1106.1445}{arXiv:1106.1445}\\
  \textit{Used in}: WRQ~4--5, CPJG Projects~CG2--CG3.\\
  \textit{Master bib key}: \texttt{Wilde2017}.\\
  \textit{Notes}: The most accessible modern treatment of quantum information theory.
  Chapter~11 (entropy inequalities) and Chapter~12 (CPTP maps) are directly relevant.

\item \textbf{[NEW]}
  \textbf{Zurek, W.~H.}
  Decoherence, einselection, and the quantum origins of the classical.
  \textit{Reviews of Modern Physics}, \textbf{75}(3), 715--775, 2003.\\
  \textit{DOI}: \href{https://doi.org/10.1103/RevModPhys.75.715}{10.1103/RevModPhys.75.715}\\
  \textit{Used in}: CPJU Project~CJ2, WRQ~2; ORQ~3.\\
  \textit{Suggested master bib key}: \texttt{Zurek2003} \textbf{[P]}.\\
  \textit{Notes}: \textbf{Add to master bibliography.} The definitive review of
  decoherence and pointer states. Sections~I--II are accessible at T3--T4;
  Section~IV (quantum Darwinism) is T5 reading. Used in CPJU~CJ2 (decoherence project).

\item \textbf{[NEW]}
  \textbf{Breuer, H.-P.\ \& Petruccione, F.}
  \textit{The Theory of Open Quantum Systems}.
  Oxford University Press, 2002.\\
  \textit{Used in}: CPJG Project~CG2; ORQ~4.\\
  \textit{Suggested master bib key}: \texttt{BreuerPetruccione2002} \textbf{[M]}.\\
  \textit{Notes}: \textbf{Add to master bibliography.} The standard graduate
  reference for open quantum systems. Chapter~3 (quantum master equations) and
  Chapter~4 (Lindblad equations) directly support CG2 and ORQ~4.
  The Born--Markov and rotating-wave approximations are derived carefully.

\item \textbf{[NEW]}
  \textbf{Plenio, M.~B.\ \& Virmani, S.}
  An introduction to entanglement measures.
  \textit{Quantum Information and Computation}, \textbf{7}(1--2), 1--51, 2007.\\
  \textit{Available free}: \href{https://arxiv.org/abs/quant-ph/0504163}{arXiv:quant-ph/0504163}\\
  \textit{Used in}: SPJG Project~SG2, WRQ~4.\\
  \textit{Suggested master bib key}: \texttt{PlenioVirmani2007} \textbf{[P]}.\\
  \textit{Notes}: \textbf{Add to master bibliography.} The clearest review of
  entanglement measures including entanglement of formation, distillable entanglement,
  negativity, and their properties. Section~4 (mixed-state measures) supports SG2.

\item \textbf{[NEW]}
  \textbf{Fawzi, O.\ \& Renner, R.}
  Quantum conditional mutual information and approximate Markov chains.
  \textit{Communications in Mathematical Physics}, \textbf{340}(2), 575--611, 2015.\\
  \textit{DOI}: \href{https://doi.org/10.1007/s00220-015-2466-x}{10.1007/s00220-015-2466-x}\\
  \textit{Used in}: CPJG Project~CG3.\\
  \textit{Suggested master bib key}: \texttt{FawziRenner2015} \textbf{[P]}.\\
  \textit{Notes}: \textbf{Add to master bibliography.} Establishes the approximate
  quantum Markov chain and recoverability from the Petz map. Key reference for CG3.

\item \textbf{[NEW]}
  \textbf{Li, H.\ \& Haldane, F.~D.~M.}
  Entanglement spectrum as a generalization of entanglement entropy.
  \textit{Physical Review Letters}, \textbf{101}(1), 010504, 2008.\\
  \textit{DOI}: \href{https://doi.org/10.1103/PhysRevLett.101.010504}{10.1103/PhysRevLett.101.010504}\\
  \textit{Used in}: WRQ~5.\\
  \textit{Suggested master bib key}: \texttt{LiHaldane2008} \textbf{[P]}.\\
  \textit{Notes}: \textbf{Add to master bibliography.} Introduces the entanglement
  spectrum (eigenvalues of $-\ln\hat{\rho}_A$) as a topological fingerprint.
  Key reference for WRQ~5 Part~(c).

\item \textbf{[NEW]}
  \textbf{White, S.~R.}
  Density matrix formulation for quantum renormalization groups.
  \textit{Physical Review Letters}, \textbf{69}(19), 2863--2866, 1992.\\
  \textit{DOI}: \href{https://doi.org/10.1103/PhysRevLett.69.2863}{10.1103/PhysRevLett.69.2863}\\
  \textit{Used in}: WRQ~5 Part~(b), CPJG Project~CG4.\\
  \textit{Suggested master bib key}: \texttt{White1992} \textbf{[P]}.\\
  \textit{Notes}: \textbf{Add to master bibliography.} The original DMRG paper.
  Explains why the reduced density matrix eigenvalues determine the optimal
  truncation basis. The physical argument (keep eigenstates with largest weight)
  is accessible at T4.

\end{enumerate}

\section{Motivation: state vs representation}
A persistent confusion is to identify a state with one of its representations: a wavefunction, a column vector,
or a matrix. In Dirac notation, the state is the abstract ket $|\psi\rangle$. A representation is obtained only
after choosing a basis, typically associated with a measurement.

\begin{warnbox}
\textbf{Pitfall:} A ket is not a column vector until a basis is chosen. Changing basis changes the coordinates,
not the physical state.
\end{warnbox}

\section{Sakurai track: preparation, outcomes, and amplitudes}
\subsection{Preparation and measurement}
A preparation is a reproducible procedure. Measuring an observable $A$ yields outcomes $a_i$ with stable frequencies,
defining probabilities $P(a_i)$.

\subsection{Why amplitudes? Interference}
Experiments show that indistinguishable alternatives do not combine by adding probabilities. Instead,
complex \emph{amplitudes} add first, then probabilities are obtained by modulus-squared.

\subsection{Born rule (operational)}
We postulate complex numbers $\langle a_i|\psi\rangle$ such that
\begin{equation}
P(a_i)=|\langle a_i|\psi\rangle|^2.
\end{equation}
Normalization follows from total probability:
\begin{equation}
\sum_i |\langle a_i|\psi\rangle|^2 = 1.
\end{equation}

\section{Dirac track: Hilbert space, rays, and inner products}
\subsection{States are rays}
Physical states are rays: $|\psi\rangle \sim c|\psi\rangle$ for any nonzero complex $c$.
In particular, global phase is irrelevant: $|\psi\rangle\sim e^{i\theta}|\psi\rangle$.

\subsection{Orthonormal basis and completeness}
Associate a measurement with an orthonormal basis $\{|n\rangle\}$:
\[
\langle m|n\rangle=\delta_{mn},\qquad \sum_n |n\rangle\langle n|=\mathbb I.
\]

\subsection{Key derivation: components are inner products}
If $|\psi\rangle=\sum_n c_n|n\rangle$, then
\[
\langle m|\psi\rangle=\sum_n c_n\langle m|n\rangle = c_m,
\]
so $c_n=\langle n|\psi\rangle$.

\subsection{Global phase invariance}
If $|\psi'\rangle=e^{i\theta}|\psi\rangle$, then
\[
|\langle n|\psi'\rangle|^2 = |e^{i\theta}\langle n|\psi\rangle|^2 = |\langle n|\psi\rangle|^2.
\]

\section{Synthesis mapping}
\begin{mapbox}
\begin{tabular}{@{}ll@{}}
\toprule
\textbf{Sakurai (operational)} & \textbf{Dirac (structural)}\\ \midrule
Preparation procedure & Ket $|\psi\rangle$ (ray in $\mathcal H$)\\
Outcome label $a_i$ & Basis ket $|a_i\rangle$\\
Amplitude & Inner product $\langle a_i|\psi\rangle$\\
Probability & $|\langle a_i|\psi\rangle|^2$\\
\bottomrule
\end{tabular}
\end{mapbox}

\section{First translation dictionary}
Discrete basis: $|\psi\rangle \leftrightarrow (c_1,\dots,c_N)^T$ where $c_n=\langle n|\psi\rangle$.\\
Continuous preview: $\psi(x)=\langle x|\psi\rangle$ (Lecture 6).

\fi

## Packet S1 — Normalization

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb,physics}
\geometry{margin=1in}
\begin{document}

\section*{L01-S1 Normalization — Problem}
Let $\{|1\rangle,|2\rangle\}$ be orthonormal. Normalize $|\psi\rangle=|1\rangle+i|2\rangle$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Compute $\langle\psi|\psi\rangle$.
\item Divide by $\sqrt{\langle\psi|\psi\rangle}$.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Interpret normalization as $\sum_i P(i)=1$.
\item Verify by recomputing the norm after normalization.
\end{itemize}

\newpage
\section*{Detailed solution}
$\langle\psi|=\langle1|-i\langle2|$. Then
\[
\langle\psi|\psi\rangle
=(\langle1|-i\langle2|)(|1\rangle+i|2\rangle)
=\langle1|1\rangle+\langle2|2\rangle=2
\]
since cross terms vanish by orthogonality. Hence
\[
|\psi\rangle_{\text{norm}}=\frac{1}{\sqrt2}\left(|1\rangle+i|2\rangle\right).
\]
Check: $\langle\psi_{\text{norm}}|\psi_{\text{norm}}\rangle=1$.
\end{document}
```

## Packet S2 — Components from inner products

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb,physics}
\geometry{margin=1in}
\begin{document}

\section*{L01-S2 Components — Problem}
Let $|\psi\rangle=\sum_n c_n|n\rangle$ in an orthonormal basis $\{|n\rangle\}$. Prove $c_m=\langle m|\psi\rangle$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Multiply the expansion by $\langle m|$ from the left.
\item Use $\langle m|n\rangle=\delta_{mn}$.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Interpret $\langle m|\psi\rangle$ as a coordinate (projection).
\item Compare to dot products in $\mathbb C^N$.
\end{itemize}

\newpage
\section*{Detailed solution}
\[
\langle m|\psi\rangle=\left\langle m\middle|\sum_n c_n|n\rangle\right\rangle
=\sum_n c_n\langle m|n\rangle
=\sum_n c_n\delta_{mn}
=c_m.
\]
\end{document}
```

## Packet S3 — Probabilities from amplitudes

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb,physics}
\geometry{margin=1in}
\begin{document}

\section*{L01-S3 Probabilities — Problem}
In an orthonormal basis, let $|\psi\rangle=\frac{1}{\sqrt5}(|1\rangle+2i|2\rangle)$. Compute $P(1)$ and $P(2)$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Compute $c_1=\langle1|\psi\rangle$, $c_2=\langle2|\psi\rangle$.
\item Use $P(n)=|c_n|^2$.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Check normalization: $P(1)+P(2)=1$.
\item Note phase does not affect modulus.
\end{itemize}

\newpage
\section*{Detailed solution}
$c_1=1/\sqrt5$ and $c_2=2i/\sqrt5$. Thus
\[
P(1)=|c_1|^2=\frac15,\qquad P(2)=|c_2|^2=\frac{4}{5}.
\]
\end{document}
```

## Packet S4 — Global phase invariance

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb,physics}
\geometry{margin=1in}
\begin{document}

\section*{L01-S4 Global phase — Problem}
Show that $P(n)=|\langle n|\psi\rangle|^2$ is invariant under $|\psi\rangle\to e^{i\theta}|\psi\rangle$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Pull out $e^{i\theta}$ using linearity in the ket.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Explain why this implies states are rays.
\item Contrast with relative phase in interference.
\end{itemize}

\newpage
\section*{Detailed solution}
\[
|\langle n|e^{i\theta}\psi\rangle|^2
=|e^{i\theta}\langle n|\psi\rangle|^2
=|e^{i\theta}|^2|\langle n|\psi\rangle|^2
=|\langle n|\psi\rangle|^2
\]
since $|e^{i\theta}|=1$.
\end{document}
```

## Packet S5 — Completeness (Parseval identity)

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb,physics}
\geometry{margin=1in}
\begin{document}

\section*{L01-S5 Completeness — Problem}
Given $\sum_n |n\rangle\langle n|=\mathbb I$, prove $\langle\psi|\psi\rangle=\sum_n|\langle n|\psi\rangle|^2$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Insert the identity between $\langle\psi|$ and $|\psi\rangle$.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Interpret as ``total probability'' in basis $\{|n\rangle\}$.
\item Recognize this as a quantum version of Parseval’s identity.
\end{itemize}

\newpage
\section*{Detailed solution}
\[
\langle\psi|\psi\rangle=\langle\psi|\mathbb I|\psi\rangle
=\left\langle\psi\middle|\sum_n|n\rangle\langle n|\middle|\psi\right\rangle
=\sum_n \langle\psi|n\rangle\langle n|\psi\rangle
=\sum_n |\langle n|\psi\rangle|^2.
\]
\end{document}
```

## Packet C1 — Cauchy–Schwarz

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb,physics}
\geometry{margin=1in}
\begin{document}

\section*{L01-C1 Cauchy--Schwarz — Problem}
Prove $|\langle\phi|\psi\rangle|^2\le \langle\phi|\phi\rangle\langle\psi|\psi\rangle$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Consider $\|\phi-\lambda\psi\|^2\ge 0$ for all $\lambda\in\mathbb C$.
\item Expand and choose $\lambda$ to minimize.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Use $\lambda=\frac{\langle\psi|\phi\rangle}{\langle\psi|\psi\rangle}$ when $\psi\neq 0$.
\item Rearrange the inequality after substitution.
\end{itemize}

\newpage
\section*{Detailed solution}
Let $f(\lambda)=\|\phi-\lambda\psi\|^2=\langle\phi-\lambda\psi|\phi-\lambda\psi\rangle\ge 0$.
Expand:
\[
f(\lambda)=\langle\phi|\phi\rangle-\lambda\langle\phi|\psi\rangle-\lambda^*\langle\psi|\phi\rangle
+|\lambda|^2\langle\psi|\psi\rangle.
\]
Choose $\lambda=\frac{\langle\psi|\phi\rangle}{\langle\psi|\psi\rangle}$. Then
\[
0\le f(\lambda)=\langle\phi|\phi\rangle-\frac{|\langle\psi|\phi\rangle|^2}{\langle\psi|\psi\rangle}.
\]
Multiply by $\langle\psi|\psi\rangle>0$:
\[
|\langle\psi|\phi\rangle|^2\le \langle\phi|\phi\rangle\langle\psi|\psi\rangle.
\]
Since $|\langle\psi|\phi\rangle|=|\langle\phi|\psi\rangle|$, we are done.
\end{document}
```

## Packet C2 — Triangle inequality

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb,physics}
\geometry{margin=1in}
\begin{document}

\section*{L01-C2 Triangle inequality — Problem}
Let $\|\psi\|=\sqrt{\langle\psi|\psi\rangle}$. Prove $\|\phi+\psi\|\le \|\phi\|+\|\psi\|$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Square both sides.
\item Expand $\|\phi+\psi\|^2$ and bound cross term using Cauchy--Schwarz.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Use $\Re z\le |z|$.
\item Conclude by taking square roots (nonnegativity).
\end{itemize}

\newpage
\section*{Detailed solution}
\[
\|\phi+\psi\|^2=\langle\phi+\psi|\phi+\psi\rangle
=\|\phi\|^2+\|\psi\|^2+2\Re\langle\phi|\psi\rangle.
\]
Since $\Re\langle\phi|\psi\rangle\le |\langle\phi|\psi\rangle|\le \|\phi\|\|\psi\|$ (C--S),
\[
\|\phi+\psi\|^2 \le \|\phi\|^2+\|\psi\|^2+2\|\phi\|\|\psi\|=(\|\phi\|+\|\psi\|)^2.
\]
Taking square roots yields $\|\phi+\psi\|\le \|\phi\|+\|\psi\|$.
\end{document}
```

## Packet C3 — Unitary change-of-basis preserves probabilities

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb}
\geometry{margin=1in}
\begin{document}

\section*{L01-C3 Unitary invariance — Problem}
Let $\mathbf c$ be coefficients of $|\psi\rangle$ in basis A. Let $\mathbf c'=U\mathbf c$ where $U^\dagger U=I$.
Show $\sum_i|c'_i|^2=\sum_i|c_i|^2$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Use vector notation: $\|\mathbf c\|^2=\mathbf c^\dagger \mathbf c$.
\item Compute $\|\mathbf c'\|^2$ and insert $U^\dagger U$.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Interpret as probability conservation under basis change.
\item Connect to inner-product preservation: $\langle U\phi|U\psi\rangle=\langle\phi|\psi\rangle$.
\end{itemize}

\newpage
\section*{Detailed solution}
\[
\sum_i|c'_i|^2=\|\mathbf c'\|^2=(U\mathbf c)^\dagger(U\mathbf c)=\mathbf c^\dagger U^\dagger U\mathbf c=\mathbf c^\dagger\mathbf c=\sum_i|c_i|^2.
\]
Thus normalization (and total probability) is preserved.
\end{document}
```

## Packet C4 — Phase-invariant distance (AdvUG/MSc)

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb,physics}
\geometry{margin=1in}
\begin{document}

\section*{L01-C4 Phase-invariant distance — Problem}
Define $d(\phi,\psi)=\min_{\theta}\|\phi-e^{i\theta}\psi\|$ for normalized states.
Show $d(\phi,\psi)$ depends only on the rays and express it in terms of $|\langle\phi|\psi\rangle|$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Expand $\|\phi-e^{i\theta}\psi\|^2$.
\item Use $\Re(e^{i\theta}\langle\phi|\psi\rangle)$ and choose $\theta$ to maximize it.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Write $\langle\phi|\psi\rangle=re^{i\alpha}$ and choose $\theta=-\alpha$.
\item Conclude $d^2=2-2|\langle\phi|\psi\rangle|$ for normalized states.
\end{itemize}

\newpage
\section*{Detailed solution}
Assume $\|\phi\|=\|\psi\|=1$. Then
\[
\|\phi-e^{i\theta}\psi\|^2=\langle\phi-e^{i\theta}\psi|\phi-e^{i\theta}\psi\rangle
=1+1- e^{i\theta}\langle\phi|\psi\rangle- e^{-i\theta}\langle\psi|\phi\rangle
=2-2\Re(e^{i\theta}\langle\phi|\psi\rangle).
\]
Let $\langle\phi|\psi\rangle=re^{i\alpha}$ with $r=|\langle\phi|\psi\rangle|$. Then
\[
\Re(e^{i\theta}\langle\phi|\psi\rangle)=\Re(re^{i(\theta+\alpha)})=r\cos(\theta+\alpha).
\]
Maximizing occurs at $\theta=-\alpha$ giving maximum $r$. Hence
\[
d^2(\phi,\psi)=2-2r=2-2|\langle\phi|\psi\rangle|.
\]
This depends only on $|\langle\phi|\psi\rangle|$, which is ray-invariant.
\end{document}
```

```latex
\section{Motivation: from measurement outcomes to operators}
\begin{sakbox}
In the lab, an \emph{observable} is a measurement procedure that outputs a real number.
If we prepare a state $|\psi\rangle$ and repeat the measurement of $A$, we obtain a distribution of outcomes.
The \emph{expectation value} is the average outcome:
\[
\langle A\rangle_\psi = \sum_i a_i\,P(a_i|\psi).
\]
We want a compact, representation-independent way to compute this.
\end{sakbox}

\section{Operators: abstract action and matrix representation}
\begin{dirbox}
An operator $\hat A$ is a linear map $\mathcal H\to\mathcal H$.
It takes a state to a new state: $|\psi\rangle \mapsto \hat A|\psi\rangle$.
\end{dirbox}

\subsection{Matrix elements and the bridge to matrices}
Choose an orthonormal basis $\{|n\rangle\}$.
Define the \emph{matrix elements} of $\hat A$ by
\[
A_{mn}=\langle m|\hat A|n\rangle.
\]
Then for any $|\psi\rangle=\sum_n c_n|n\rangle$, we have
\[
\hat A|\psi\rangle=\sum_{m,n} |m\rangle\,A_{mn}\,c_n,
\]
so the coefficient vector transforms as
\[
\mathbf c' = A\,\mathbf c,
\]
where $A$ is the matrix with entries $A_{mn}$.
This is exactly matrix mechanics.

\begin{warnbox}
A matrix is a \emph{representation} of an operator in a chosen basis.
Changing basis changes the matrix, not the abstract operator.
\end{warnbox}

\subsection{Expectation values in bra--ket and in matrices}
Define
\[
\langle \hat A\rangle_\psi = \langle\psi|\hat A|\psi\rangle.
\]
In the basis $\{|n\rangle\}$ this becomes
\[
\langle\psi|\hat A|\psi\rangle
= \sum_{m,n} c_m^* A_{mn} c_n
= \mathbf c^\dagger A \mathbf c.
\]

\section{Adjoint and Hermitian operators}
The adjoint $\hat A^\dagger$ is defined by
\[
\langle\phi|\hat A|\psi\rangle = \langle \hat A^\dagger \phi|\psi\rangle
\quad\text{for all }|\phi\rangle,|\psi\rangle.
\]
In a basis, $(A^\dagger)_{mn}=A_{nm}^*$, i.e. $A^\dagger$ is the conjugate transpose matrix.

\subsection{Hermitian (self-adjoint) and reality of measurement outcomes}
An observable corresponds to a Hermitian operator (physics shorthand) $\hat A=\hat A^\dagger$.
Then
\[
\langle\psi|\hat A|\psi\rangle \in \mathbb R.
\]
Proof (short): take complex conjugate and use Hermiticity.

\section{Unitary operators: symmetries and basis changes}
A unitary operator $\hat U$ satisfies $\hat U^\dagger\hat U=I$.
It preserves inner products:
\[
\langle U\phi|U\psi\rangle = \langle\phi|\psi\rangle,
\]
so it preserves norms and probabilities.

\subsection{Basis change vs physical transformation}
If we \emph{change basis} by a unitary $U$, the matrix of an operator transforms by similarity:
\[
A' = UAU^\dagger.
\]
If we \emph{physically transform the state}, we write
\[
|\psi'\rangle = U|\psi\rangle.
\]
Both involve $U$, but they represent different conceptual moves.

\begin{mapbox}
\begin{tabular}{@{}ll@{}}
\toprule
\textbf{Abstract} & \textbf{Matrix in basis } $\{|n\rangle\}$\\ \midrule
$|\psi\rangle$ & column vector $\mathbf c$ with $c_n=\langle n|\psi\rangle$\\
$\hat A$ & matrix $A$ with $A_{mn}=\langle m|\hat A|n\rangle$\\
$\hat A|\psi\rangle$ & $\mathbf c' = A\mathbf c$\\
$\langle\psi|\hat A|\psi\rangle$ & $\mathbf c^\dagger A \mathbf c$\\
$\hat A^\dagger$ & conjugate transpose $A^\dagger$\\
unitary $\hat U$ & $U^\dagger U=I$\\
\bottomrule
\end{tabular}
\end{mapbox}

\section{Worked example: 2D system and Pauli matrices}
Let $\{|0\rangle,|1\rangle\}$ be a basis and consider
\[
\sigma_x=\begin{pmatrix}0&1\\1&0\end{pmatrix},\quad
\sigma_z=\begin{pmatrix}1&0\\0&-1\end{pmatrix}.
\]
These are Hermitian. For $|\psi\rangle=\alpha|0\rangle+\beta|1\rangle$,
\[
\langle\sigma_z\rangle = |\alpha|^2-|\beta|^2.
\]
Projector onto $|0\rangle$ is $P_0=|0\rangle\langle0|$ with matrix $\begin{pmatrix}1&0\\0&0\end{pmatrix}$.
```

## **S1 — Matrix elements from bra–ket**

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb,physics}
\geometry{margin=1in}
\begin{document}

\section*{L03-S1 Matrix elements — Problem}
Let $\{|0\rangle,|1\rangle\}$ be an orthonormal basis. Define
\[
\hat A = 2|0\rangle\langle 0| + (1+i)|0\rangle\langle 1| + (1-i)|1\rangle\langle 0| - |1\rangle\langle 1|.
\]
Compute the matrix $A_{mn}=\langle m|\hat A|n\rangle$ in this basis.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Use $\langle m|\,|a\rangle\langle b|\,|n\rangle=\langle m|a\rangle\langle b|n\rangle$.
\item Orthonormality gives $\langle m|n\rangle=\delta_{mn}$.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Recognize that $|0\rangle\langle1|$ contributes only to the $(0,1)$ entry.
\item Write the matrix by inspecting each dyad $|m\rangle\langle n|$.
\end{itemize}

\newpage
\section*{Detailed solution}
Compute entries:
\[
A_{00}=\langle0|\hat A|0\rangle=2,\quad
A_{01}=\langle0|\hat A|1\rangle=1+i,
\]
\[
A_{10}=\langle1|\hat A|0\rangle=1-i,\quad
A_{11}=\langle1|\hat A|1\rangle=-1.
\]
Thus
\[
A=\begin{pmatrix}2 & 1+i\\ 1-i & -1\end{pmatrix}.
\]
\end{document}
```

## **S2 — Operator action as matrix multiplication**

```latex

\section*{L03-S2 Action — Problem}
Let $A=\begin{pmatrix}2 & 1+i\\ 1-i & -1\end{pmatrix}$ and $c=\begin{pmatrix}\alpha\\ \beta\end{pmatrix}$.
Compute $c'=Ac$ and interpret $c'$ as the coefficients of $\hat A|\psi\rangle$ in the basis.

\begin{itemize}

\[
c'=\begin{pmatrix}2 & 1+i\\ 1-i & -1\end{pmatrix}\begin{pmatrix}\alpha\\ \beta\end{pmatrix}

\]
These are the coefficients of $\hat A|\psi\rangle$ in the chosen basis.
```

## **S3 — Expectation value (bra–ket ↔ matrix)**

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb}
\geometry{margin=1in}
\begin{document}

\section*{L03-S3 Expectation value — Problem}
Let $A=\begin{pmatrix}2 & 1+i\\ 1-i & -1\end{pmatrix}$ and normalized $c=\begin{pmatrix}\alpha\\ \beta\end{pmatrix}$.
Compute $\langle A\rangle=c^\dagger A c$ explicitly.

\newpage
\section*{Suggestions}
\begin{itemize}
\item First compute $Ac$.
\item Then compute $c^\dagger(Ac)$.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Write $c^\dagger=(\alpha^*,\beta^*)$.
\item Keep track of complex conjugates.
\end{itemize}

\newpage
\section*{Detailed solution}
\[
Ac=\begin{pmatrix}2\alpha+(1+i)\beta\\ (1-i)\alpha-\beta\end{pmatrix}.
\]
Then
\[
\langle A\rangle = (\alpha^*,\beta^*)\begin{pmatrix}2\alpha+(1+i)\beta\\ (1-i)\alpha-\beta\end{pmatrix}
=2|\alpha|^2-\;|\beta|^2 + \alpha^*(1+i)\beta + \beta^*(1-i)\alpha.
\]
Note the last two terms are complex conjugates, so their sum is $2\Re\!\big((1+i)\alpha^*\beta\big)$:
\[
\boxed{\langle A\rangle = 2|\alpha|^2-|\beta|^2 + 2\Re\!\big((1+i)\alpha^*\beta\big).}
\]
\end{document}
```

## **S4 — Adjoint and adjoint property**

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb}
\geometry{margin=1in}
\begin{document}

\section*{L03-S4 Adjoint — Problem}
Let $A=\begin{pmatrix}0 & 2+i\\ 1 & 3i\end{pmatrix}$.
(i) Compute $A^\dagger$. (ii) Verify for vectors $x,y\in\mathbb C^2$ that $x^\dagger(Ay)=(A^\dagger x)^\dagger y$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item $A^\dagger$ is conjugate transpose.
\item Use associativity of matrix multiplication.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Write $x^\dagger(Ay)=(x^\dagger A)y$ and compare with $(A^\dagger x)^\dagger y$.
\end{itemize}

\newpage
\section*{Detailed solution}
(i) Transpose and conjugate:
\[
A^\dagger=\begin{pmatrix}0^* & 1^*\\ (2+i)^* & (3i)^*\end{pmatrix}
=\begin{pmatrix}0 & 1\\ 2-i & -3i\end{pmatrix}.
\]
(ii) Using matrix identities, $(A^\dagger x)^\dagger = x^\dagger A$ (since $(A^\dagger x)^\dagger=x^\dagger A$).
Thus
\[
(A^\dagger x)^\dagger y = (x^\dagger A)y = x^\dagger(Ay),
\]
which verifies the adjoint property.
\end{document}
```

```latex
\section*{Advanced suggestions}
\begin{itemize}
\item Note that $c^\dagger A c$ equals its own complex conjugate if $A=A^\dagger$.
\end{itemize}

\newpage
\section*{Detailed solution}
(i) $A^\dagger=\begin{pmatrix}2 & (1-i)^*\\ (1+i)^* & -1\end{pmatrix}=\begin{pmatrix}2 & 1+i\\ 1-i & -1\end{pmatrix}=A$.
(ii) From S3:
\[
\langle A\rangle = 2|\alpha|^2-|\beta|^2 + \alpha^*(1+i)\beta + \beta^*(1-i)\alpha.
\]
The last two terms are complex conjugates, hence their sum is $2\Re((1+i)\alpha^*\beta)$.
So $\langle A\rangle\in\mathbb R$.
\end{document}
```

## **S6 — Unitary check + norm preservation**

```latex

\usepackage{geometry,amsmath,amssymb}
\geometry{margin=1in}
\begin{document}

\section*{L03-S6 Unitary — Problem}
Let $U=\frac{1}{\sqrt2}\begin{pmatrix}1&1\\1&-1\end{pmatrix}$.
(i) Show $U^\dagger U=I$. (ii) Show $\|Uc\|=\|c\|$ for any $c\in\mathbb C^2$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Compute $U^\dagger$ and multiply.
\item Use $\|c\|^2=c^\dagger c$.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item For (ii): $(Uc)^\dagger(Uc)=c^\dagger(U^\dagger U)c$.
\end{itemize}

\newpage
\section*{Detailed solution}
(i) $U$ is real symmetric, so $U^\dagger=U^T=U$. Then
\[
U^\dagger U = \frac{1}{2}\begin{pmatrix}1&1\\1&-1\end{pmatrix}\begin{pmatrix}1&1\\1&-1\end{pmatrix}
=\frac{1}{2}\begin{pmatrix}2&0\\0&2\end{pmatrix}=I.
\]
(ii)
\[
\|Uc\|^2=(Uc)^\dagger(Uc)=c^\dagger(U^\dagger U)c=c^\dagger c=\|c\|^2.
\]
```

## **S9 — Basis change $$A'=UAU^\dagger$$ in 2×2**

```latex

\usepackage{geometry,amsmath,amssymb}
\geometry{margin=1in}

Let $A=\begin{pmatrix}1&0\\0&-1\end{pmatrix}$ and $U=\frac{1}{\sqrt2}\begin{pmatrix}1&1\\1&-1\end{pmatrix}$.
Compute $A'=UAU^\dagger$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Note $U^\dagger=U$ here.
\item Multiply step by step: first $B=UA$, then $A'=BU$.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Interpret $A'$ as the matrix of the same abstract operator in the new basis.
\end{itemize}

\newpage
\section*{Detailed solution}
Compute:
\[
UA=\frac{1}{\sqrt2}\begin{pmatrix}1&1\\1&-1\end{pmatrix}\begin{pmatrix}1&0\\0&-1\end{pmatrix}
=\frac{1}{\sqrt2}\begin{pmatrix}1&-1\\1&1\end{pmatrix}.
\]
Then
\[
A'= (UA)U = \frac{1}{2}\begin{pmatrix}1&-1\\1&1\end{pmatrix}\begin{pmatrix}1&1\\1&-1\end{pmatrix}
=\frac{1}{2}\begin{pmatrix}0&2\\2&0\end{pmatrix}
=\begin{pmatrix}0&1\\1&0\end{pmatrix}.
\]
So $A'=\sigma_x$.
```

## **C1 — Hermitian ⇒ real expectation (proof)**

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb,physics}
\geometry{margin=1in}

\section*{L03-C1 Hermitian expectation — Problem}
Prove that if $\hat A=\hat A^\dagger$, then for any $|\psi\rangle$, $\langle\psi|\hat A|\psi\rangle\in\mathbb R$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Take complex conjugate of $\langle\psi|\hat A|\psi\rangle$.
\item Use $(\langle\psi|\hat A|\psi\rangle)^*=\langle\psi|\hat A^\dagger|\psi\rangle$.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Show the quantity equals its own conjugate.
\end{itemize}

\newpage
\section*{Detailed solution}
Let $z=\langle\psi|\hat A|\psi\rangle$. Then
\[
z^* = (\langle\psi|\hat A|\psi\rangle)^* = \langle\psi|\hat A^\dagger|\psi\rangle
\]
(using the defining property of the adjoint). If $\hat A=\hat A^\dagger$, then $z^*=z$.
A complex number equal to its conjugate is real. Hence $z\in\mathbb R$.
```

## **C3 — Basis-change law $$A'=UAU^\dagger$$ (proof)**

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb}
\geometry{margin=1in}
\begin{document}

\section*{L03-C3 Basis-change transformation — Problem}
Let $|n'\rangle=\sum_m U_{mn}|m\rangle$ with $U$ unitary. Show that the matrix of $\hat A$ in the primed basis is
$A'=UAU^\dagger$, i.e. $A'_{ij}=\langle i'|\hat A|j'\rangle$ equals $(UAU^\dagger)_{ij}$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Write $\langle i'|=\sum_p U_{pi}^*\langle p|$.
\item Substitute into $\langle i'|\hat A|j'\rangle$ and rearrange sums.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Recognize $\sum_{p,q} U_{pi}^* A_{pq} U_{qj}$ as $(U^\dagger A U)_{ij}$ or $(UAU^\dagger)_{ij}$ depending on convention; check indices carefully.
\end{itemize}

\newpage
\section*{Detailed solution}
We compute
\[
A'_{ij}=\langle i'|\hat A|j'\rangle
=\left(\sum_p U_{pi}^*\langle p|\right)\hat A\left(\sum_q U_{qj}|q\rangle\right)
=\sum_{p,q} U_{pi}^*\,\langle p|\hat A|q\rangle\,U_{qj}.
\]
But $A_{pq}=\langle p|\hat A|q\rangle$, so
\[
A'_{ij}=\sum_{p,q} (U^\dagger)_{ip}\,A_{pq}\,U_{qj}=(U^\dagger A U)_{ij}.
\]
If you define the basis change oppositely (new coords $c'=Uc$), you obtain $A'=UAU^\dagger$.
Both are consistent once you fix the convention. The invariant statement is: the operator matrix transforms by unitary similarity.
\end{document}
```

## **C4 — Projector algebra + interpretation**

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb,physics}
\geometry{margin=1in}
\begin{document}

\section*{L03-C4 Projectors — Problem}
Let $P=|a\rangle\langle a|$ with $\langle a|a\rangle=1$.
(i) Prove $P^2=P$ and $P^\dagger=P$.
(ii) Show that for any $|\psi\rangle$, $P|\psi\rangle=|a\rangle\langle a|\psi\rangle$ is the component of $|\psi\rangle$ along $|a\rangle$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Use dyad multiplication and normalization.
\item Interpret $\langle a|\psi\rangle$ as a scalar coefficient.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Show $P$ acts as identity on the span of $|a\rangle$ and annihilates orthogonal vectors.
\end{itemize}

\newpage
\section*{Detailed solution}
(i) Same as S7:
\[
P^2=|a\rangle(\langle a|a\rangle)\langle a|=|a\rangle\langle a|=P,\quad
P^\dagger=|a\rangle\langle a|=P.
\]
(ii)
\[
P|\psi\rangle = |a\rangle\langle a|\psi\rangle.
\]
This is a scalar multiple of $|a\rangle$, hence lies in the subspace spanned by $|a\rangle$.
If $|\psi\rangle$ is orthogonal to $|a\rangle$, then $\langle a|\psi\rangle=0$ and $P|\psi\rangle=0$.
So $P$ extracts the $|a\rangle$ component.
\end{document}
```

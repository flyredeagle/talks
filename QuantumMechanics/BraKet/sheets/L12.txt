```latex
\section{Why foundations matter}
Dirac notation is powerful because it is representation-independent. But it can hide delicate issues:
\begin{itemize}
\item Continuous bases involve distributions: $\langle x|x'\rangle=\delta(x-x')$.
\item Important observables ($x$, $p$, $H$) are unbounded operators: domains matter.
\item ``Hermitian'' in physics often means ``self-adjoint''; these differ for unbounded operators.
\end{itemize}

\begin{warnbox}
Treat $\delta$ as a distribution and $|x\rangle$ as a generalized object. Formal manipulations are safe only when
they can be justified by inserting test functions and checking boundary/domain assumptions.
\end{warnbox}

\section{Physics-rigor track: safe Dirac rules}
\begin{physbox}
\subsection*{Rulebook (practical)}
\begin{enumerate}[leftmargin=*]
\item \textbf{Always test against a wavefunction.} Replace expressions with their action on $\psi(x)=\langle x|\psi\rangle$.
\item \textbf{Use completeness as an operator identity.} $\int dx\,|x\rangle\langle x|=I$ should be applied inside inner products.
\item \textbf{Use delta only under integrals.} $\delta$ is defined by $\int f(x)\delta(x-a)\,dx=f(a)$ for nice $f$.
\item \textbf{Track boundary terms.} Derivations like $\hat p=-i\hbar\partial_x$ assume suitable decay or boundary conditions.
\item \textbf{Check normalization/positivity.} Probabilities must be $\ge0$ and sum/integrate to 1.
\item \textbf{Separate ``formal'' from ``physical.''} $|x\rangle$ is idealized; physical states are square-integrable wavepackets.
\end{enumerate}

\subsection*{Example: completeness produces inner product}
\[
\langle\phi|\psi\rangle = \int dx\,\langle\phi|x\rangle\langle x|\psi\rangle=\int dx\,\phi^*(x)\psi(x).
\]

\subsection*{Example: momentum operator in $x$-rep}
From $\langle x|p\rangle\propto e^{ipx/\hbar}$, we obtain $(\hat p\psi)(x)=-i\hbar\partial_x\psi(x)$,
assuming we can exchange differentiation and integration and boundary terms vanish.
\end{physbox}

\section{Math-forward track: Hilbert spaces, domains, self-adjointness}
\begin{mathbox}
\subsection*{Hilbert space and domains}
A Hilbert space $\mathcal H$ is a complete inner-product space. Many quantum observables are \emph{unbounded},
so they cannot be defined on all of $\mathcal H$. They have a dense domain $D(A)\subset\mathcal H$.

\subsection*{Adjoint and self-adjointness}
Given a densely-defined operator $A$, the adjoint $A^\dagger$ is defined via
\[
\langle \phi|A\psi\rangle=\langle A^\dagger \phi|\psi\rangle\quad \text{for }\psi\in D(A),\ \phi\in D(A^\dagger).
\]
An operator is \textbf{symmetric} if $\langle \phi|A\psi\rangle=\langle A\phi|\psi\rangle$ on $D(A)$.
It is \textbf{self-adjoint} if $A=A^\dagger$ and $D(A)=D(A^\dagger)$.

\subsection*{Why self-adjoint matters}
Self-adjointness guarantees real spectrum and generates unitary evolution $U(t)=e^{-itA/\hbar}$ (Stone's theorem).

\end{mathbox}

\section{Rigged Hilbert spaces and generalized eigenvectors}
To handle continuous spectra, one uses a Gelfand triple (rigged Hilbert space):
\[
\Phi \subset \mathcal H \subset \Phi',
\]
where $\Phi$ is a dense test-function space (e.g. Schwartz functions), and $\Phi'$ its dual (distributions).
Generalized eigenvectors like $|x\rangle$ live in $\Phi'$ rather than $\mathcal H$.

\section{Spectral measures (PVM) viewpoint}
A self-adjoint operator $A$ has a spectral measure $E(\lambda)$ such that
\[
A=\int \lambda\, dE(\lambda),
\]
and probabilities for outcome in a set $\Delta$ are
\[
p(\Delta)=\langle\psi|E(\Delta)|\psi\rangle.
\]
Discrete case reduces to sums with projectors; continuous case uses integrals.

\begin{mapbox}
\begin{tabular}{@{}ll@{}}
\toprule
Formal Dirac & Safe interpretation\\ \midrule
$|x\rangle$ & distributional functional on test functions\\
$\delta(x-x')$ & defined by action under integrals\\
$A(x,x')=\langle x|A|x'\rangle$ & integral kernel on $\psi(x')$\\
Hermitian & symmetric on a domain (physics shorthand)\\
Self-adjoint & correct observable notion; real spectrum; unitary group\\
\bottomrule
\end{tabular}
\end{mapbox}
```

## **S1 — Inner product via completeness**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath,amssymb,physics}\geometry{margin=1in}\begin{document}
\section*{L12-S1 Inner product from completeness — Problem}
Assume $\int dx\,|x\rangle\langle x|=I$. Show $\langle\phi|\psi\rangle=\int dx\,\phi^*(x)\psi(x)$.

\newpage\section*{Suggestions}\begin{itemize}
\item Insert the identity between $\langle\phi|$ and $|\psi\rangle$.
\end{itemize}

\newpage\section*{Advanced suggestions}\begin{itemize}
\item Use $\langle\phi|x\rangle=\phi^*(x)$ and $\langle x|\psi\rangle=\psi(x)$.
\end{itemize}

\newpage\section*{Detailed solution}
$\langle\phi|\psi\rangle=\int dx\,\langle\phi|x\rangle\langle x|\psi\rangle=\int dx\,\phi^*(x)\psi(x)$.
\end{document}
```

## **S2 — Delta sifting**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath}\geometry{margin=1in}\begin{document}
\section*{L12-S2 Delta sifting — Problem}
Evaluate $\int_{-\infty}^{\infty} dx\, (1+x)\delta(x-2)$.

\newpage\section*{Suggestions}\begin{itemize}
\item Apply $\int f(x)\delta(x-a)dx=f(a)$.
\end{itemize}

\newpage\section*{Advanced suggestions}\begin{itemize}
\item State that $\delta$ is a distribution defined by this property on test functions.
\end{itemize}

$f(x)=1+x$, $a=2$ so the integral equals $f(2)=3$.
```

## **S3 — Delta reproducing**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath}\geometry{margin=1in}\begin{document}
\section*{L12-S3 Delta reproducing — Problem}
Compute $\int dx\,\delta(x-x')g(x)$ and interpret the result.

\newpage\section*{Suggestions}\begin{itemize}
\item Treat $x'$ as fixed and apply sifting.
```

## **S4 — Kernel action**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath}\geometry{margin=1in}\begin{document}
\section*{L12-S4 Kernel action — Problem}
Let $A(x,x')=\delta(x-x')x'^2$. Compute $(A\psi)(x)=\int dx'\,A(x,x')\psi(x')$.

\newpage\section*{Suggestions}\begin{itemize}
\item Use delta to collapse the integral.
\end{itemize}

\newpage\section*{Advanced suggestions}\begin{itemize}
\item Note $x'^2$ becomes $x^2$ after delta acts.
\end{itemize}

\newpage\section*{Detailed solution}
\[
(A\psi)(x)=\int dx'\,\delta(x-x')x'^2\psi(x')=x^2\psi(x).
\]
\end{document}
```

## **S5 — Position operator kernel**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath,amssymb,physics}\geometry{margin=1in}\begin{document}
\section*{L12-S5 Kernel of $\hat x$ (formal) — Problem}
Show that the kernel of $\hat x$ in the $x$ basis is $x\delta(x-x')$, i.e. $\langle x|\hat x|x'\rangle=x\delta(x-x')$.

\newpage\section*{Suggestions}\begin{itemize}
\item Use $\hat x|x'\rangle=x'|x'\rangle$ and take $\langle x|$.
\end{itemize}

\newpage\section*{Advanced suggestions}\begin{itemize}
\item Replace $x'$ by $x$ under $\delta(x-x')$ when acting under integrals.
\end{itemize}

\newpage\section*{Detailed solution}
$\langle x|\hat x|x'\rangle=\langle x|x' x'\rangle=x'\langle x|x'\rangle=x'\delta(x-x')$.
Under integrals this acts as $x\delta(x-x')$, so we write $x\delta(x-x')$ (formal equivalence).
\end{document}
```

## **S6 — Momentum operator kernel**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath,amssymb}\geometry{margin=1in}\begin{document}
\section*{L12-S6 Kernel of $\hat p$ (formal) — Problem}
Assuming $(\hat p\psi)(x)=-i\hbar\partial_x\psi(x)$, show a formal kernel is $A(x,x')=-i\hbar\,\partial_x\delta(x-x')$.

\newpage\section*{Suggestions}\begin{itemize}
\item Start from $(A\psi)(x)=\int dx'\,A(x,x')\psi(x')$ and aim to produce a derivative of $\psi$.
\end{itemize}

\newpage\section*{Advanced suggestions}\begin{itemize}
\item Use distribution identity: $\int dx'\,\partial_x\delta(x-x')\psi(x')=\partial_x\psi(x)$.
\end{itemize}

\newpage\section*{Detailed solution}
Formally,
\[
\int dx'\,\partial_x\delta(x-x')\psi(x')=\partial_x\int dx'\,\delta(x-x')\psi(x')=\partial_x\psi(x).
\]
Thus choosing $A(x,x')=-i\hbar\,\partial_x\delta(x-x')$ yields $(A\psi)(x)=-i\hbar\,\partial_x\psi(x)$.
\end{document}
```

## **S7 — Symmetry of momentum with decay (integration by parts)**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath,amssymb}\geometry{margin=1in}\begin{document}
\section*{L12-S7 Symmetry of $-i\hbar\partial_x$ — Problem}
Let $(p\psi)(x)=-i\hbar\psi'(x)$. Show $\langle\phi|p\psi\rangle=\langle p\phi|\psi\rangle$ assuming $\phi,\psi$ vanish at $\pm\infty$.

\newpage\section*{Suggestions}\begin{itemize}
\item Write $\langle\phi|p\psi\rangle=\int dx\,\phi^*(x)(-i\hbar\psi'(x))$.
\item Integrate by parts.
\end{itemize}

\newpage\section*{Advanced suggestions}\begin{itemize}
\item Boundary term $[-i\hbar\phi^*\psi]_{-\infty}^{\infty}$ vanishes by assumption.
\end{itemize}

\newpage\section*{Detailed solution}
\[
\int dx\,\phi^*(-i\hbar\psi') = -i\hbar\left[\phi^*\psi\right]_{-\infty}^{\infty}+i\hbar\int dx\,(\phi^*)'\psi.
\]
Boundary term vanishes, and $(\phi^*)'=(\phi')^*$, hence
\[
\langle\phi|p\psi\rangle=\int dx\,(i\hbar\phi')^*\psi=\langle p\phi|\psi\rangle.
\]
So $p$ is symmetric on this domain.
\end{document}
```

## **S8 — Boundary terms do not vanish on an interval**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath}\geometry{margin=1in}\begin{document}
\section*{L12-S8 Boundary counterexample — Problem}
On $[0,L]$, let $(p\psi)(x)=-i\hbar\psi'(x)$ with no boundary conditions. Show integration by parts yields a boundary term and explain why symmetry can fail.

\newpage\section*{Suggestions}\begin{itemize}
\item Repeat integration by parts but evaluate boundary term at $0$ and $L$.
\end{itemize}

\newpage\section*{Advanced suggestions}\begin{itemize}
\item Show symmetry requires $\phi^*(L)\psi(L)-\phi^*(0)\psi(0)=0$ for all $\phi,\psi$ in the domain.
\end{itemize}

\newpage\section*{Detailed solution}
\[
\int_0^L dx\,\phi^*(-i\hbar\psi') = -i\hbar[\phi^*\psi]_0^L + i\hbar\int_0^L dx\,(\phi^*)'\psi.
\]
The boundary term is $-i\hbar(\phi^*(L)\psi(L)-\phi^*(0)\psi(0))$ and need not vanish.
Without boundary conditions restricting the domain, symmetry can fail because $\langle\phi|p\psi\rangle\neq\langle p\phi|\psi\rangle$.
\end{document}
```

## **S9 — Self-adjoint implies real expectation**

```latex
\section*{L12-C2 Unboundedness and domains — Problem}
Give a careful argument (1 page) why $x$ and $p$ are unbounded on $L^2(\mathbb R)$ and thus cannot be defined on all of $\mathcal H$ as bounded operators.

\newpage\section*{Suggestions}
\begin{itemize}

\item Similar for $p$ using high-frequency oscillations.
\end{itemize}

\newpage\section*{Advanced suggestions}
\begin{itemize}
\item Mention: bounded operator would satisfy $\|A\psi\|\le C\|\psi\|$ for all $\psi$; show no such $C$ exists.
\end{itemize}

For $x$: choose normalized $\psi_n$ concentrated near $x=n$ (shifted wavepackets). Then $\|x\psi_n\|\approx n\|\psi_n\|=n$ grows without bound, so no uniform constant $C$ exists.
Similarly for $p=-i\hbar\partial_x$: choose $\psi_n(x)=e^{inx}\phi(x)$ with fixed normalized $\phi$; then $\|p\psi_n\|$ grows like $n$, showing unboundedness.
Therefore domains must be restricted to vectors for which $x\psi$ and $p\psi$ remain square-integrable.
```

## **C4 — Spectral calculus outline**

```latex
\documentclass[11pt]{article}\usepackage{geometry}\geometry{margin=1in}\begin{document}
\section*{L12-C4 Spectral calculus — Problem}
Given the spectral form $A=\int \lambda\,dE(\lambda)$, outline why for suitable $f$ one has
$f(A)=\int f(\lambda)\,dE(\lambda)$ and how this generalizes polynomial functional calculus.

\newpage\section*{Suggestions}
\begin{itemize}
\item Start with polynomials: if $A=\sum aP_a$, then $f(A)=\sum f(a)P_a$.
\item Replace the sum by an integral over the spectral measure.
\end{itemize}

\newpage\section*{Advanced suggestions}
\begin{itemize}
\item Mention approximating continuous $f$ by polynomials or simple functions.
\item Note $E(\Delta)$ are projectors and obey measure properties.
\end{itemize}

\newpage\section*{Detailed solution (outline)}
In the discrete case, $A=\sum_a aP_a$ implies $A^n=\sum_a a^n P_a$ and hence $f(A)=\sum_a f(a)P_a$ for polynomials.
For general spectra, one defines $f(A)$ first for simple functions $f=\sum_k c_k \chi_{\Delta_k}$ by $f(A)=\sum_k c_k E(\Delta_k)$ and extends by limits to broader classes of $f$.
This yields $f(A)=\int f(\lambda)\,dE(\lambda)$, consistent with the discrete sum when the measure has atoms at eigenvalues.
\end{document}
```

## **C5 — Stone theorem structured sketch**

```latex
\documentclass[11pt]{article}\usepackage{geometry}\geometry{margin=1in}\begin{document}
\section*{L12-C5 Stone theorem sketch — Problem}
Write a structured sketch explaining:
(i) what a strongly continuous one-parameter unitary group $U(t)$ is,
(ii) why it implies the existence of a self-adjoint generator $H$ with $U(t)=e^{-itH/\hbar}$,
(iii) how this connects to physical Hamiltonians.

\newpage\section*{Suggestions}
\begin{itemize}
\item Mention group property $U(t+s)=U(t)U(s)$ and strong continuity.
\item State the theorem (no full proof).
\end{itemize}

\newpage\section*{Advanced suggestions}
\begin{itemize}
\item Explain why self-adjointness (not just symmetry) is essential for true unitary evolution.
\end{itemize}

\newpage\section*{Detailed solution (sketch)}
A strongly continuous unitary group satisfies $U(0)=I$, $U(t+s)=U(t)U(s)$, and $\lim_{t\to0}\|U(t)\psi-\psi\|=0$ for all $\psi$.
Stone's theorem states there exists a unique self-adjoint operator $H$ such that $U(t)=e^{-itH/\hbar}$.
Physically, $H$ is the Hamiltonian generating time evolution; self-adjointness ensures real energy spectrum and conservation of probability through unitarity.
\end{document}
```

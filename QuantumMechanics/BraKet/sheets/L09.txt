```latex
\section{Composite systems and tensor products}
If system $A$ has Hilbert space $\mathcal H_A$ and system $B$ has $\mathcal H_B$, the composite system is described by
\[
\mathcal H_{AB}=\mathcal H_A\otimes\mathcal H_B.
\]
A product basis is $\{|i\rangle_A\otimes|j\rangle_B\}$, often abbreviated $|i\rangle|j\rangle$.

\subsection{Inner products factorize on product states}
For product kets $|a\rangle|b\rangle$ and $|c\rangle|d\rangle$,
\[
(\langle a|\langle b|)(|c\rangle|d\rangle)=\langle a|c\rangle\,\langle b|d\rangle.
\]

\section{Local operators}
An operator acting only on $A$ is represented as $\hat A\otimes I_B$.
Similarly, acting only on $B$ is $I_A\otimes \hat B$.
On a product state,
\[
(\hat A\otimes I)(|a\rangle|b\rangle)=(\hat A|a\rangle)|b\rangle.
\]

\begin{warnbox}
\textbf{Pitfall:} $(\hat A\otimes \hat B)(|a\rangle|b\rangle)=(\hat A|a\rangle)\otimes(\hat B|b\rangle)$, not $(\hat A\hat B)|ab\rangle$.
Tensor products are not ordinary multiplication.
\end{warnbox}

\section{Entanglement}
A pure state $|\Psi\rangle\in\mathcal H_A\otimes\mathcal H_B$ is \emph{separable} if it can be written as
\[
|\Psi\rangle=|\psi\rangle_A\otimes|\phi\rangle_B.
\]
Otherwise it is \emph{entangled}.
Example (Bell state):
\[
|\Phi^+\rangle=\frac{1}{\sqrt2}(|00\rangle+|11\rangle).
\]
No choice of single-system kets $|\psi\rangle,|\phi\rangle$ reproduces these joint amplitudes by factorization.

\section{Density matrix and reduced state (partial trace preview)}
For a pure state $|\Psi\rangle$, define
\[
\rho_{AB}=|\Psi\rangle\langle\Psi|.
\]
To describe statistics of measurements performed only on $A$, we use the reduced state
\[
\rho_A=\Tr_B(\rho_{AB}),
\]
the \emph{partial trace over $B$}. Operationally, this corresponds to ``ignoring'' or ``discarding'' system $B$.

\subsection{Defining the partial trace}
Let $\{|j\rangle_B\}$ be an orthonormal basis for $B$. Then
\[
\Tr_B(\rho_{AB})=\sum_j \left(I_A\otimes\langle j|\right)\rho_{AB}\left(I_A\otimes|j\rangle\right).
\]
This definition is basis-independent (a key property).

\subsection{Local expectation values from reduced states}
For any operator $A$ on system $A$,
\[
\langle A\otimes I\rangle_{\rho_{AB}}=\Tr\left(\rho_{AB}(A\otimes I)\right)=\Tr\left(\rho_A A\right).
\]
Thus $\rho_A$ contains all information needed to predict measurement statistics on $A$ alone.

\subsection{Worked example: Bell state reduction}
For $|\Phi^+\rangle$, one finds
\[
\rho_A=\frac12 I,\qquad \rho_B=\frac12 I,
\]
so each subsystem looks maximally mixed even though the joint state is pure and perfectly correlated.

\section{Schmidt decomposition (preview)}
Any bipartite pure state can be written as
\[
|\Psi\rangle=\sum_{k=1}^r \sqrt{\lambda_k}\,|u_k\rangle_A|v_k\rangle_B,
\]
where $\lambda_k\ge 0$, $\sum_k\lambda_k=1$, and $\{|u_k\rangle\}$, $\{|v_k\rangle\}$ are orthonormal.
The state is entangled iff more than one $\lambda_k$ is nonzero. Reduced states satisfy
\[
\rho_A=\sum_k \lambda_k |u_k\rangle\langle u_k|,\qquad
\rho_B=\sum_k \lambda_k |v_k\rangle\langle v_k|.
\]

\begin{mapbox}
\begin{tabular}{@{}lll@{}}
\toprule
\textbf{Dirac} & \textbf{Matrix/Kronecker} & \textbf{Reduced state}\\ \midrule
$|\psi\rangle\otimes|\phi\rangle$ & $\mathbf c\otimes \mathbf d$ & $\rho_A=\Tr_B\rho_{AB}$\\
$A\otimes I$ & $A\otimes I$ (Kronecker) & $\Tr(\rho_{AB}(A\otimes I))=\Tr(\rho_A A)$\\
Bell state $|\Phi^+\rangle$ & vector $(1,0,0,1)/\sqrt2$ & $\rho_A=I/2$\\
\bottomrule
\end{tabular}
\end{mapbox}
```

## **S1 — Tensor product inner product**

```latex
\documentclass[11pt]{article}
\usepackage{geometry,amsmath,amssymb,physics}
\geometry{margin=1in}
\begin{document}

\section*{L09-S1 Inner product on product states — Problem}
Show $(\langle a|\otimes\langle b|)(|c\rangle\otimes|d\rangle)=\langle a|c\rangle\,\langle b|d\rangle$.

\newpage
\section*{Suggestions}
\begin{itemize}
\item Use the definition of tensor product action on linear functionals.
\end{itemize}

\newpage
\section*{Advanced suggestions}
\begin{itemize}
\item Verify bilinearity in each slot; check on basis vectors and extend by linearity.
\end{itemize}

\newpage
\section*{Detailed solution}
By definition of the tensor product of linear maps, $(f\otimes g)(u\otimes v)=f(u)g(v)$.
Taking $f=\langle a|$, $g=\langle b|$, $u=|c\rangle$, $v=|d\rangle$ gives
\[
(\langle a|\otimes\langle b|)(|c\rangle\otimes|d\rangle)=\langle a|c\rangle\,\langle b|d\rangle.
\]
\end{document}
```

## **S2 — Expand state and compute joint probabilities**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath,amssymb}\geometry{margin=1in}\begin{document}
\section*{L09-S2 Joint probabilities — Problem}
Let $|\Psi\rangle=\frac{1}{\sqrt3}(|00\rangle+|01\rangle+|10\rangle)$ in the computational basis.
Compute $P(00),P(01),P(10),P(11)$.

\newpage\section*{Suggestions}\begin{itemize}
\item Probabilities are squared magnitudes of basis amplitudes.

\newpage\section*{Advanced suggestions}\begin{itemize}
```

## **S3 — Local operator action**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath,amssymb,physics}\geometry{margin=1in}\begin{document}
\section*{L09-S3 Local operator — Problem}
Show $(A\otimes I)(|a\rangle|b\rangle)=(A|a\rangle)|b\rangle$.

\newpage\section*{Suggestions}\begin{itemize}
\item Use definition: $(A\otimes I)(u\otimes v)=(Au)\otimes(Iv)$.
\end{itemize}

\newpage\section*{Advanced suggestions}\begin{itemize}
\item Extend to superpositions by linearity.
\end{itemize}

\newpage\section*{Detailed solution}
By definition of tensor product operators,
\[
(A\otimes I)(|a\rangle\otimes|b\rangle)=(A|a\rangle)\otimes(I|b\rangle)=(A|a\rangle)\otimes|b\rangle.

\end{document}
```


## **S7 — Density matrix from a pure state**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath,amssymb}\geometry{margin=1in}\begin{document}
\section*{L09-S7 Build $\rho_{AB}$ — Problem}
Let $|\Psi\rangle=\frac{1}{\sqrt3}(|00\rangle+|01\rangle+|10\rangle)$. Write $\rho_{AB}=|\Psi\rangle\langle\Psi|$ as a $4\times4$ matrix in the basis $(|00\rangle,|01\rangle,|10\rangle,|11\rangle)$.

\end{document}
```

## **S8 — Partial trace for Bell state**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath,amssymb,physics}\geometry{margin=1in}\begin{document}
\section*{L09-S8 Partial trace preview — Problem}
Let $|\Phi^+\rangle=(|00\rangle+|11\rangle)/\sqrt2$ and $\rho_{AB}=|\Phi^+\rangle\langle\Phi^+|$.
Compute $\rho_A=\Tr_B(\rho_{AB})$.

\newpage\section*{Suggestions}\begin{itemize}
\item Use basis $\{|0\rangle_B,|1\rangle_B\}$ and $\Tr_B(\rho)=\sum_{j}(I\otimes\langle j|)\rho(I\otimes|j\rangle)$.
\end{itemize}

\newpage\section*{Advanced suggestions}\begin{itemize}
\item Expand $\rho_{AB}$ as a sum of four outer products and trace term-by-term.
\end{itemize}

\newpage\section*{Detailed solution}
First
\[
\rho_{AB}=\frac12\Big(|00\rangle\langle00|+|00\rangle\langle11|+|11\rangle\langle00|+|11\rangle\langle11|\Big).
\]
Trace over $B$ in basis $\{|0\rangle,|1\rangle\}$.
Only terms with matching $B$ bra/ket survive:
\[
\Tr_B(|00\rangle\langle00|)=|0\rangle\langle0|,\quad
\Tr_B(|11\rangle\langle11|)=|1\rangle\langle1|.
\]
Cross terms vanish because $\langle0|1\rangle=0$:
\[
\Tr_B(|00\rangle\langle11|)=0,\quad \Tr_B(|11\rangle\langle00|)=0.
\]
Thus
\[
\rho_A=\frac12\left(|0\rangle\langle0|+|1\rangle\langle1|\right)=\frac12 I.
\]
\end{document}
```

## **S10 — Verify local statistics identity**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath,amssymb,physics}\geometry{margin=1in}\begin{document}

Let $\rho_{AB}=|\Phi^+\rangle\langle\Phi^+|$ and $A=\sigma_z$. Verify

by computing both sides.
```

## **C1 — Extend factorization by linearity**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath,amssymb}\geometry{margin=1in}\begin{document}
\section*{L09-C1 Bilinearity extension — Problem}
Prove that the inner product on $\mathcal H_A\otimes\mathcal H_B$ defined on product states by
$\langle a,b|c,d\rangle=\langle a|c\rangle\langle b|d\rangle$ extends uniquely by linearity to all finite superpositions.

\newpage\section*{Suggestions}\begin{itemize}
\item Use linearity in each argument and the distributive law.
\item Define for $\sum_i u_i\otimes v_i$ and $\sum_j u'_j\otimes v'_j$ by double sum.
\end{itemize}

\newpage\section*{Advanced suggestions}\begin{itemize}
\item Show well-definedness: the value does not depend on a particular decomposition once the tensor product is formed (outline-level).
\end{itemize}

\newpage\section*{Detailed solution}
Let $|\Psi\rangle=\sum_i |u_i\rangle|v_i\rangle$ and $|\Phi\rangle=\sum_j |u'_j\rangle|v'_j\rangle$.
Define
\[
\langle\Phi|\Psi\rangle := \sum_{i,j}\langle u'_j|u_i\rangle\langle v'_j|v_i\rangle,
\]
which is sesquilinear and matches the product rule on pure tensors. In finite-dimensional Hilbert spaces this is the standard construction and yields a valid inner product.
\end{document}
```

## **C2 — Basis-independence of partial trace**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath,amssymb,physics}\geometry{margin=1in}\begin{document}
\section*{L09-C2 Basis-independence of partial trace — Problem}
Show $\rho_A=\Tr_B(\rho_{AB})$ is independent of the orthonormal basis chosen for tracing over $B$.

\newpage\section*{Suggestions}\begin{itemize}
\item Let $\{|j\rangle\}$ and $\{|j'\rangle\}$ be ONBs related by a unitary $W$.
\item Express $|j'\rangle=\sum_k W_{kj}|k\rangle$ and substitute.
\end{itemize}

\newpage\section*{Advanced suggestions}\begin{itemize}
\item Use $\sum_j |j'\rangle\langle j'|=I$ and the cyclicity of trace in the $B$ slot.
\end{itemize}

\newpage\section*{Detailed solution}
Define $\Tr_B$ by $\Tr_B(X)=\sum_j (I\otimes\langle j|)X(I\otimes|j\rangle)$.
If $|j'\rangle=W|j\rangle$ with unitary $W$, then
\[
\sum_{j'}(I\otimes\langle j'|)X(I\otimes|j'\rangle)
=\sum_j (I\otimes\langle j|W^\dagger)X(I\otimes W|j\rangle)
=\sum_j (I\otimes\langle j|)\,(I\otimes W^\dagger)X(I\otimes W)\,(I\otimes|j\rangle).
\]
But $(I\otimes W^\dagger)X(I\otimes W)$ is a unitary change of basis on subsystem $B$; tracing over $B$ removes dependence on that basis because $\sum_j |j\rangle\langle j|=I$ and the trace is invariant under unitary similarity on $B$. Hence the result is basis-independent.
\end{document}
```

## **C3 — Defining property of partial trace**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath,amssymb,physics}\geometry{margin=1in}\begin{document}
\section*{L09-C3 Partial trace property — Problem}
Prove that for any $\rho_{AB}$ and any operator $A$ on subsystem $A$,
\[
\Tr(\rho_{AB}(A\otimes I))=\Tr(\rho_A A),
\quad \rho_A=\Tr_B(\rho_{AB}).
\]

\newpage\section*{Suggestions}\begin{itemize}
\item Use a basis $\{|j\rangle\}$ for $B$.
\item Insert $I=\sum_j I\otimes|j\rangle\langle j|$ appropriately and use trace cyclicity.
\end{itemize}

\newpage\section*{Advanced suggestions}\begin{itemize}
\item Write $\Tr_{AB}=\Tr_A\circ\Tr_B$ and use the definition of $\Tr_B$.
\end{itemize}

\newpage\section*{Detailed solution}
Let $\{|j\rangle\}$ be an ONB of $B$.
By definition,
\[
\rho_A=\Tr_B(\rho_{AB})=\sum_j (I\otimes\langle j|)\rho_{AB}(I\otimes|j\rangle).
\]
Then
\[
\Tr(\rho_A A)=\Tr_A\!\left(\sum_j (I\otimes\langle j|)\rho_{AB}(I\otimes|j\rangle)\,A\right).
\]
Using $\Tr_{AB}(X)=\Tr_A\left(\sum_j (I\otimes\langle j|)X(I\otimes|j\rangle)\right)$, set $X=\rho_{AB}(A\otimes I)$:
\[
\Tr_{AB}(\rho_{AB}(A\otimes I))=\Tr_A\!\left(\sum_j (I\otimes\langle j|)\rho_{AB}(A\otimes I)(I\otimes|j\rangle)\right).
\]
Because $(A\otimes I)(I\otimes|j\rangle)=A\otimes|j\rangle$, this equals $\Tr(\rho_A A)$, proving the identity.
\end{document}
```

## **C4 — Schmidt decomposition (2-qubit computation)**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath,amssymb}\geometry{margin=1in}\begin{document}
\section*{L09-C4 Schmidt decomposition — Problem}
Let $|\Psi\rangle=\sqrt{0.8}\,|00\rangle+\sqrt{0.2}\,|11\rangle$.
(i) Identify Schmidt coefficients.
(ii) Compute $\rho_A$ and its eigenvalues.

\newpage\section*{Suggestions}\begin{itemize}
\item The state is already in Schmidt form with $|u_1\rangle=|0\rangle$, $|v_1\rangle=|0\rangle$ etc.
\item Compute $\rho_{AB}=|\Psi\rangle\langle\Psi|$ then trace out $B$.
\end{itemize}

\newpage\section*{Advanced suggestions}\begin{itemize}
\item Cross terms vanish in partial trace because $\langle0|1\rangle=0$ on $B$.
\end{itemize}

\newpage\section*{Detailed solution}
(i) Schmidt coefficients are $\lambda_1=0.8$, $\lambda_2=0.2$.
(ii) $\rho_A=\Tr_B\rho_{AB}=0.8\,|0\rangle\langle0|+0.2\,|1\rangle\langle1|$.
Eigenvalues are $0.8$ and $0.2$.
\end{document}
```

## **C5 — Entanglement entropy equality (outline + example)**

```latex
\documentclass[11pt]{article}\usepackage{geometry,amsmath}\geometry{margin=1in}\begin{document}
\section*{L09-C5 Entanglement entropy equality — Problem}
For a pure bipartite state $\rho_{AB}=|\Psi\rangle\langle\Psi|$, outline why $S(\rho_A)=S(\rho_B)$ and verify for $|\Psi\rangle=\sqrt{0.8}|00\rangle+\sqrt{0.2}|11\rangle$.

\newpage\section*{Suggestions}
\begin{itemize}
\item Use Schmidt decomposition: reduced states share the same nonzero eigenvalues.
\item Entropy depends only on eigenvalues.
\end{itemize}

\newpage\section*{Advanced suggestions}
\begin{itemize}
\item For the example, compute $S=-\sum_k \lambda_k\log\lambda_k$ using $\{0.8,0.2\}$.
\end{itemize}

\newpage\section*{Detailed solution}
Schmidt decomposition implies $\rho_A=\sum_k \lambda_k |u_k\rangle\langle u_k|$ and $\rho_B=\sum_k \lambda_k |v_k\rangle\langle v_k|$ share the same eigenvalues $\{\lambda_k\}$ (plus zeros if dimensions differ). Thus $S(\rho_A)=S(\rho_B)$.
Example: eigenvalues are $0.8,0.2$ so
\[
S(\rho_A)=-(0.8\log 0.8 + 0.2\log 0.2)=S(\rho_B).
\]
\end{document}
```
